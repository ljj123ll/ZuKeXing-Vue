  <!-- 登录页面容器：用于整体布局 -->

### 脚本部分（Script Setup）
```typescript
<script setup lang="ts">
// 导入Vue核心API：ref（创建响应式基本类型）、reactive（创建响应式对象）
import { ref, reactive } from 'vue';
// 导入Vue Router：用于路由跳转
import { useRouter } from 'vue-router';
// 导入Element Plus组件：用于类型定义（如表单实例类型）
import { ElForm, ElFormItem, ElInput, ElButton, ElMessage } from 'element-plus';
// 导入登录接口：login是请求函数，LoginParams是接口参数类型
import { login, LoginParams } from '@/apis/user';
// 导入用户状态管理：用于保存登录信息（如token、用户信息）
import { useUserStore } from '@/stores/modules/user';

// 表单引用：创建响应式变量，类型为ElForm组件实例（用于调用表单验证等方法）
const loginFormRef = ref<InstanceType<typeof ElForm>>();

// 加载状态：响应式变量，控制登录按钮的加载动画（true表示正在加载）
const loading = ref<boolean>(false);

// 登录表单数据：创建响应式对象，类型为LoginParams（约束account和password字段）
const loginForm = reactive<LoginParams>({
  account: '', // 账号初始值为空
  password: '' // 密码初始值为空
});

// 表单验证规则：响应式对象，定义account和password的验证逻辑
const loginRules = reactive({
  account: [
    // 规则1：必填验证
    { required: true, message: '用户名不能为空', trigger: ['blur', 'submit'] }, 
    // 规则2：长度验证（5-15个字符）
    { min: 5, max: 15, message: '账号长度在 5 到 15 个字符', trigger: 'blur' }
  ],
  password: [
    // 规则1：必填验证
    { required: true, message: '密码不能为空', trigger: ['blur', 'submit'] }, 
    // 规则2：长度验证（6-16个字符）
    { min: 6, max: 16, message: '密码长度在 6 到 16 个字符', trigger: 'blur' }
  ]
});

// 路由实例：创建路由对象，用于登录成功后跳转页面
const router = useRouter();

// 用户状态实例：创建用户状态对象，用于保存登录信息
const userStore = useUserStore();

// 登录处理函数：异步函数（支持await语法）
const handleLogin = async () => {
  // 若表单引用不存在，直接返回（避免报错）
  if (!loginFormRef.value) return;
  
  try {
    // 验证表单：调用Element Plus表单的validate方法，若验证失败会抛出错误
    await loginFormRef.value.validate();
    // 开始加载：设置loading为true（按钮显示加载动画）
    loading.value = true;
    // 调用登录接口：传入表单数据，等待接口返回结果
    const response = await login(loginForm);
    
    // 登录成功处理：
    // 1. 保存登录信息到状态管理（token和用户信息）
    userStore.saveLoginInfo(response.result.token, response.result.userInfo);
    // 2. 显示成功提示
    ElMessage.success('登录成功');
    // 3. 延迟500ms后跳转到首页（等待提示动画完成）
    setTimeout(() => router.push('/'), 500);
  } catch (error: any) {
    // 捕获错误：打印错误信息到控制台
    console.error('登录失败:', error);
    // 忽略表单验证错误（Element Plus会自动显示验证提示）
    if (error.name === 'ValidationError') return;
    
    // 处理接口错误：
    const errorMsg = error.message || ''; // 获取错误信息（默认空字符串）
    // 若错误包含"账户不存在"或"密码错误"，统一提示"账号或密码错误"（避免暴露具体错误）
    if (errorMsg.includes('账户不存在') || errorMsg.includes('密码错误')) {
      ElMessage.error('账号或密码错误');
    } else {
      // 其他错误（如服务器异常、账号禁用等）显示原始错误信息
      ElMessage.error(errorMsg || '登录失败，请稍后重试');
    }
  } finally {
    // 无论成功/失败，最终都关闭加载状态
    loading.value = false;
  }
};
</script>
```

### 总结
这个组件是一个完整的登录页面，功能包括：
1. 表单布局：使用Element Plus组件实现账号、密码输入框和登录按钮
2. 表单验证：验证账号/密码的必填性和长度限制
3. 登录逻辑：调用登录接口，处理成功/失败状态，保存登录信息
4. 交互体验：加载动画、回车登录、错误提示、成功跳转
5. 样式美化：居中布局、卡片设计、阴影效果等视觉优化
